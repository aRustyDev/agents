## Error Handling

### Exponential Backoff Strategy

```python
BACKOFF_CONFIG = {
    "initial_delay": 2,      # seconds
    "multiplier": 2,
    "max_delay": 300,        # 5 minute ceiling
    "max_retries": 5,
}

def fetch_with_backoff(url: str, session: httpx.Client) -> httpx.Response | None:
    """Fetch with exponential backoff until ceiling, then log and continue."""
    delay = BACKOFF_CONFIG["initial_delay"]

    for attempt in range(BACKOFF_CONFIG["max_retries"]):
        try:
            response = session.get(url, timeout=30)

            if response.status_code == 429:  # Rate limited
                retry_after = int(response.headers.get("Retry-After", delay))
                logger.warning(f"Rate limited, waiting {retry_after}s (attempt {attempt + 1})")
                time.sleep(retry_after)
                continue

            if response.status_code >= 500:  # Server error
                logger.warning(f"Server error {response.status_code}, backoff {delay}s")
                time.sleep(delay)
                delay = min(delay * BACKOFF_CONFIG["multiplier"], BACKOFF_CONFIG["max_delay"])
                continue

            return response

        except (httpx.TimeoutException, httpx.ConnectError) as e:
            logger.warning(f"Connection error: {e}, backoff {delay}s (attempt {attempt + 1})")
            time.sleep(delay)
            delay = min(delay * BACKOFF_CONFIG["multiplier"], BACKOFF_CONFIG["max_delay"])

    # Max retries exhausted - log failure and continue
    logger.error(f"FAILED after {BACKOFF_CONFIG['max_retries']} attempts: {url}")
    log_failure(url, "max_retries_exhausted")
    return None
```

### Partial Failure Handling

On any failure:
1. **Write partial results** to output files (append mode for NDJSON)
2. **Log failure point** to `.crawl-state.json` with URL, timestamp, error
3. **Execute exponential backoff** up to 5-minute ceiling
4. **Log final failure** if backoff exhausted
5. **Continue to next item** - don't abort crawl

```python
def handle_failure(url: str, error: str, state: CrawlState):
    """Log failure and continue crawling."""
    state.failures.append({
        "url": url,
        "error": error,
        "timestamp": datetime.utcnow().isoformat(),
        "will_retry": True,
    })
    state.save()
    # Continue - don't raise
```

### Resume Logic

```python
def resume_crawl(tier: str, state: CrawlState):
    """Resume from last checkpoint."""
    tier_state = state.tiers.get(tier, {})

    for registry, reg_state in tier_state.get("registries", {}).items():
        if reg_state["status"] == "completed":
            logger.info(f"Skipping {registry} (already completed)")
            continue

        if reg_state["status"] == "in_progress":
            logger.info(f"Resuming {registry} from page {reg_state.get('last_page', 1)}")
            # Resume from checkpoint

        # Also retry any failures from this registry
        for failure in state.failures:
            if registry in failure["url"] and failure["will_retry"]:
                logger.info(f"Retrying failed URL: {failure['url']}")
```

### Daily Progress Tracking

```python
class DailyRateLimiter:
    """Track daily request counts for rate-limited APIs."""

    def __init__(self, registry: str, daily_limit: int):
        self.registry = registry
        self.daily_limit = daily_limit
        self.requests_today = 0
        self.reset_date = date.today()

    def can_request(self) -> bool:
        if date.today() > self.reset_date:
            self.requests_today = 0
            self.reset_date = date.today()
        return self.requests_today < self.daily_limit

    def record_request(self):
        self.requests_today += 1
        if self.requests_today >= self.daily_limit:
            logger.warning(f"{self.registry}: Daily limit reached ({self.daily_limit})")
            logger.info(f"Resume tomorrow or run: --tier api --resume")
```

### Validation Script

```python
def validate_output(output_dir: Path) -> bool:
    """Validate all output files against schemas."""
    schema_dir = output_dir / "schemas"

    validators = {
        "components.json": schema_dir / "components.schema.json",
        "registries.json": schema_dir / "registries.schema.json",
        "plugin_components.json": schema_dir / "plugin_components.schema.json",
    }

    all_valid = True
    for data_file, schema_file in validators.items():
        data_path = output_dir / data_file
        if not data_path.exists():
            logger.warning(f"Missing: {data_file}")
            continue

        schema = json.loads(schema_file.read_text())
        validator = jsonschema.Draft202012Validator(schema)

        # Handle NDJSON vs JSON array
        if data_file == "components.json":
            for i, line in enumerate(data_path.open()):
                doc = json.loads(line)
                errors = list(validator.iter_errors(doc))
                if errors:
                    logger.error(f"{data_file}:{i+1} - {errors[0].message}")
                    all_valid = False
        else:
            data = json.loads(data_path.read_text())
            for i, doc in enumerate(data):
                errors = list(validator.iter_errors(doc))
                if errors:
                    logger.error(f"{data_file}[{i}] - {errors[0].message}")
                    all_valid = False

    return all_valid
```
